```csharp
public bool IsValidEmail(string email)
{
 // Return true if it is in valid email format.
 return System.Text.RegularExpressions.Regex.IsMatch( email,
 @"^([\w-\.]+)@(([[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.)|(([\w-]+\.)+))([a-zA-Z]{2,4}|[0-9]{1,3})(\]?)$";
}
```
